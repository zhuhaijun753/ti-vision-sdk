    44    /*!
    45     *  ======== Wugen ========
    46     *  Wugen Module
    47     *
    48     *  Using Wugen will install a Hwi beginHook that automatically clears
    49     *  the corresponding pending WUGEN IRQ bit for the interrupting event.
    50     *
    51     *  Currently this functionality is provided only for non-combined events.
    52     *
    53     *  Combined events must be cleared manually by the user's ISR.
    54     *
    55     *  Unconstrained Functions
    56     *  All functions
    57     *
    58     *  @p(html)
    59     *  <h3> Calling Context </h3>
    60     *  <table border="1" cellpadding="3">
    61     *    <colgroup span="1"></colgroup> <colgroup span="5" align="center"></colgroup>
    62     *
    63     *    <tr><th> Function                 </th><th>  Hwi   </th><th>  Swi   </th><th>  Task  </th><th>  Main  </th><th>  Startup  </th></tr>
    64     *    <!--                                                                                                                 -->
    65     *    <tr><td> {@link #applyMasks}    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    66     *    <tr><td> {@link #disableDma}     </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    67     *    <tr><td> {@link #disableEvent}    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    68     *    <tr><td> {@link #enableDma}   </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    69     *    <tr><td> {@link #enableEvent}   </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    70     *    <tr><td> {@link #getMasks}        </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    71     *    <tr><td> {@link #ierToMasks} </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    72     *    <tr><td colspan="6"> Definitions: <br />
    73     *       <ul>
    74     *         <li> <b>Hwi</b>: API is callable from a Hwi thread. </li>
    75     *         <li> <b>Swi</b>: API is callable from a Swi thread. </li>
    76     *         <li> <b>Task</b>: API is callable from a Task thread. </li>
    77     *         <li> <b>Main</b>: API is callable during any of these phases: </li>
    78     *           <ul>
    79     *             <li> In your module startup after this module is started (e.g. Mod_Module_startupDone() returns TRUE). </li>
    80     *             <li> During xdc.runtime.Startup.lastFxns. </li>
    81     *             <li> During main().</li>
    82     *             <li> During BIOS.startupFxns.</li>
    83     *           </ul>
    84     *         <li> <b>Startup</b>: API is callable during any of these phases:</li>
    85     *           <ul>
    86     *             <li> During xdc.runtime.Startup.firstFxns.</li>
    87     *             <li> In your module startup before this module is started (e.g. Mod_Module_startupDone() returns FALSE).</li>
    88     *           </ul>
    89     *       </ul>
    90     *    </td></tr>
    91     *
    92     *  </table>
    93     *  @p
    94     */
    95    
    96    @DirectCall
    97    @ModuleStartup
    98    
    99    module Wugen
   100    {
   101        /*!
   102         *  ======== applyMasks ========
   103         *  This function takes an array of 5 integers and applies the specified
   104         *  mask to the WUGEN interrupt mask registers. If the bit in the mask is
   105         *  enabled, the corresponding interrupt in the WUGEN will be blocked.
   106         *  
   107         *  @param(mask)    array of five integers for applying to the WUGEN
   108         *                  interrupt mask register.
   109         *                  mask[0] consists of IRQ 0-31
   110         *                  mask[1] consists of IRQ 32-63
   111         *                  mask[2] consists of IRQ 64-71
   112         *                  mask[3] consists of DMA 0-31 requests
   113         *                  mask[4] consists of DMA 32-63 requests
   114         */
   115        Void applyMasks(UInt32 mask[]);
   116    
   117        /*!
   118         *  ======== disableDma ========
   119         *  Disables the specified DMA source in the WUGEN
   120         *
   121         *  @param(dmaid)   the DMA id for a particular EDMA source.
   122         */
   123        Void disableDma(UInt dmaid);
   124    
   125        /*!
   126         *  ======== disableEvent ========
   127         *  Disables the specified interrupt event source in the WUGEN
   128         *
   129         *  @param(eventid) the event number of an interrupt source
   130         */
   131        Void disableEvent(UInt eventid);
   132    
   133        /*!
   134         *  ======== enableDma ========
   135         *  Enables the specified DMA source in the WUGEN
   136         *
   137         *  @param(dmaid)   the DMA id for a particular EDMA source.
   138         */
   139        Void enableDma(UInt dmaid);
   140    
   141        /*!
   142         *  ======== enableEvent ========
   143         *  Enables the specified interrupt event source in the WUGEN
   144         *
   145         *  @param(eventid) the event number of an interrupt source
   146         */
   147        Void enableEvent(UInt eventid);
   148    
   149        /*!
   150         *  ======== getMasks ========
   151         *  This function returns the WUGEN interrupt mask registers
   152         *
   153         *  @param(mask)    array of five integers used to return the WUGEN
   154         *                  interrupt mask registers
   155         *                  mask[0] consists of IRQ 0-31
   156         *                  mask[1] consists of IRQ 32-63
   157         *                  mask[2] consists of IRQ 64-71
   158         *                  mask[3] consists of DMA 0-31 requests
   159         *                  mask[4] consists of DMA 32-63 requests
   160         */
   161        Void getMasks(UInt32 mask[]);
   162    
   163        /*!
   164         *  ======== ierToMasks ========
   165         *  Compute the WUGEN masks needed to allow the interrupts set in the IER
   166         *  register to propagate through the WUGEN.  This function does not enable
   167         *  external DMA requests.  You must set these explicitly to configure a
   168         *  DMA request as a wakeup event.
   169         *
   170         *  @param(mask)    array of five integers for returning WUGEN masks.
   171         *                  mask[0] consists of IRQ 0-31
   172         *                  mask[1] consists of IRQ 32-63
   173         *                  mask[2] consists of IRQ 64-71
   174         *                  mask[3] consists of DMA 0-31 requests
   175         *                  mask[4] consists of DMA 32-63 requests
   176         */
   177        Void ierToMasks(UInt32 mask[]);
   178    
   179    internal:
   180    
   181        /*!
   182         *  ======== hwiRegisterHook ========
   183         *  Hwi register hook to capture hook ID
   184         */
   185        Void hwiRegisterHook(Int id);
   186    
   187        /*!
   188         *  ======== hwiCreateHook ========
   189         *  Hwi create hook that calculates corresponding irqNum
   190         */
   191        Void hwiCreateHook(IHwi.Handle hwi, Error.Block *eb);
   192    
   193        /*!
   194         *  ======== hwiBeginHook ========
   195         * Hwi begin hook that clears wugen pending irqNum
   196         */
   197        Void hwiBeginHook(IHwi.Handle hwi);
   198    
   199        /*!
   200         *  ======== intToIrq ========
   201         *  Convert interrupt number to IRQ number
   202         */
   203        UInt intToIrq(UInt intNum);
   204    
   205        /*!
   206         *  ======== evtToIrq ========
   207         *  Convert eventId number to IRQ number
   208         */
   209        UInt evtToIrq(UInt intNum);
   210    
   211        /*!
   212         *  ======== intToEvt ========
   213         *  Convert interrupt number to eventId
   214         */
   215        UInt intToEvt(UInt intNum);
   216    
   217        struct Module_State {
   218            Int     hwiHookId;
   219        };
   220    }
